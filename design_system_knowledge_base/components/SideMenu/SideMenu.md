## Component Name
SideMenu

## Description
The SideMenu component is a hierarchical navigation element used in sidebars. It can represent a main menu item or a nested sub-menu item and supports various interactive and semantic states (e.g., 'Active', 'Disabled', 'Error').

## 1. TypeScript Types (The "What")
This is the props interface for the SideMenu component. The AI must adhere to this structure.

```JSX
import React, { HTMLAttributes } from 'react';

// The props interface MUST extend the standard div attributes.
export interface SideMenuProps extends HTMLAttributes<HTMLDivElement> {
  /**
   * The text label for the menu item.
   */
  label: string;
  /**
   * The type of the menu item.
   * @default 'menu'
   */
  type?: 'menu' | 'sub-menu';
  /**
   * Whether the menu item has a nested sub-menu.
   * @default false
   */
  hasSubMenu?: boolean;
  /**
   * The visual state of the component.
   * @default 'Enabled'
   */
  status?: 'Enabled' | 'Hover' | 'Active' | 'Disabled' | 'Focus' | 'Error' | 'Warning' | 'Success';
  /**
   * An array of sub-menu items if 'hasSubMenu' is true.
   */
  subMenuItems?: SideMenuProps[];
}
```
## Example
Follow this exact structure. The component MUST be able to render its own `subMenuItems` recursively.

```JSX
import React, { useState } from 'react';
import styles from './SideMenu.module.css';
import clsx from 'clsx';

export interface SideMenuProps extends React.HTMLAttributes<HTMLDivElement> {
  label: string;
  type?: 'menu' | 'sub-menu';
  hasSubMenu?: boolean;
  status?: 'Enabled' | 'Active' | 'Disabled' | 'Success';
  subMenuItems?: SideMenuProps[];
}

const SideMenu: React.FC<SideMenuProps> = ({
  label,
  type = 'menu',
  hasSubMenu = false,
  status = 'Enabled',
  subMenuItems = [],
  ...props
}) => {
  // A sub-menu is open if its parent is active
  const [isOpen, setIsOpen] = useState(status === 'Active');

  const containerClassName = clsx(
    styles.side_menu_item,
    styles[type], // .menu or .sub_menu
    styles[status.toLowerCase()] // .active, .disabled, .success etc.
  );
  
  const handleToggle = () => {
    if (status !== 'Disabled' && hasSubMenu) {
      setIsOpen(!isOpen);
    }
  };

  return (
    <div className={styles.menu_item_wrapper}>
      <div className={containerClassName} onClick={handleToggle} {...props}>
        <span className={styles.icon}>ICON</span> {/* Placeholder for icon */}
        <span className={styles.label}>{label}</span>
        {hasSubMenu && (
          <span className={clsx(styles.arrow, { [styles.open]: isOpen })}>â€º</span>
        )}
      </div>
      {/* Recursive rendering of sub-menu items */}
      {isOpen && hasSubMenu && (
        <div className={styles.sub_menu_container}>
          {subMenuItems.map((item, index) => (
            <SideMenu key={index} {...item} />
          ))}
        </div>
      )}
    </div>
  );
};

export default SideMenu;

```
